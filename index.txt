    // Smooth scrolling for nav links and CTA button
    function scrollToSection(sectionId) {
      const el = document.getElementById(sectionId);
      if (el) {
        window.scrollTo({
          top: el.getBoundingClientRect().top + window.scrollY - (window.innerWidth < 700 ? 60 : 0),
          behavior: "smooth"
        });
      }
    }
    // Navigation click events
    document.querySelectorAll('.nav-item').forEach(item => {
      item.addEventListener('click', e => {
        e.preventDefault();
        const target = item.getAttribute('data-target');
        if (target) {
          scrollToSection(target);
          document.querySelectorAll('.nav-item').forEach(i => i.classList.remove('active'));
          item.classList.add('active');
        }
      });
      // Accessibility: Enter key scroll
      item.addEventListener('keydown', e => {
        if (e.key === 'Enter' || e.key === ' ') {
          e.preventDefault();
          item.click();
        }
      });
    });
    // CTA Button scroll to contact
    document.getElementById('ctaBtn').addEventListener('click', function() {
      scrollToSection('contact');
    });

    // Highlight nav based on scroll position
    const sectionIds = ["home", "about", "practice-areas", "contact"];
    function updateNavHighlight() {
      let closest = sectionIds[0];
      let minDist = Number.POSITIVE_INFINITY;
      sectionIds.forEach(id => {
        const el = document.getElementById(id);
        if (el) {
          const rect = el.getBoundingClientRect();
          const dist = Math.abs(rect.top - (window.innerWidth < 700 ? 60 : 0));
          if (dist < minDist) {
            minDist = dist;
            closest = id;
          }
        }
      });
      document.querySelectorAll('.nav-item').forEach(item => {
        item.classList.toggle('active', item.getAttribute('data-target') === closest);
      });
    }
    window.addEventListener('scroll', updateNavHighlight);
    window.addEventListener('resize', updateNavHighlight);
    document.addEventListener('DOMContentLoaded', updateNavHighlight);

    // Contact form validation
    const form = document.getElementById('contactForm');
    form.addEventListener('submit', function(e) {
      e.preventDefault();
      let isValid = true;

      // Helper: show error
      function showError(id, msg) {
        const err = document.getElementById('error-' + id);
        if (err) {
          err.textContent = msg;
          err.style.display = 'block';
        }
      }
      // Helper: hide error
      function hideError(id) {
        const err = document.getElementById('error-' + id);
        if (err) err.style.display = 'none';
      }
      function validateEmail(email) {
        // Simple regex for email validation
        return /^[^\s@]+@[^\s@]+\.[^\s@]+$/.test(email.trim());
      }
      // Validate Name
      const name = form.name.value.trim();
      if (!name) {
        showError('name', 'Name is required.');
        isValid = false;
      } else {
        hideError('name');
      }
      // Validate Email
      const email = form.email.value.trim();
      if (!email) {
        showError('email', 'Email is required.');
        isValid = false;
      } else if (!validateEmail(email)) {
        showError('email', 'Please enter a valid email address.');
        isValid = false;
      } else {
        hideError('email');
      }
      // Validate Message
      const message = form.message.value.trim();
      if (!message) {
        showError('message', 'Message cannot be empty.');
isValid = false;
      } else {
        hideError('message');
      }
      // If valid, simulate success
      const successMsg = document.getElementById('form-success');
      if (isValid) {
        successMsg.textContent = "Thank you for reaching out! Your message has been sent.";
        successMsg.style.display = "block";
        form.reset();
        setTimeout(() => { successMsg.style.display = "none"; }, 3500);
      } else {
        successMsg.style.display = "none";
      }
    });
  </script>
